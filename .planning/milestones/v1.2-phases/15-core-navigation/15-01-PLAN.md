Replace the entire file with a `'use client'` component:

```typescript
'use client'

import { useEffect, useState } from 'react'
import { Badge } from '@/components/ui/badge'

interface WebhookError {
  timestamp: string
  error: string
  context: string
}

export function WebhookFehlerBadge() {
  const [errorCount, setErrorCount] = useState(0)

  useEffect(() => {
    async function fetchErrors() {
      try {
        const res = await fetch('/api/globals/webhook_errors')
        if (!res.ok) return
        const data = await res.json()
        const errors: WebhookError[] = Array.isArray(data.errors) ? data.errors : []
        const twentyFourHoursAgo = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString()
        const recent = errors.filter(
          (e) => e.timestamp && e.timestamp > twentyFourHoursAgo
        )
        setErrorCount(recent.length)
      } catch {
        // Silently fail — nav should still work
      }
    }
    fetchErrors()
    const interval = setInterval(fetchErrors, 60_000)
    return () => clearInterval(interval)
  }, [])

  if (errorCount === 0) return null
  return <Badge variant="destructive">{errorCount}</Badge>
}

export default WebhookFehlerBadge
```
