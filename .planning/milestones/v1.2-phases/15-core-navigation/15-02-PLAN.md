**Active link detection function:**
```typescript
function isActive(pathname: string, href: string): boolean {
  if (href === '/admin') {
    return pathname === '/admin' || pathname === '/admin/'
  }
  return pathname.startsWith(href) &&
    (pathname.length === href.length || pathname[href.length] === '/')
}
```

**Helper to check if any link in a section is active** (for auto-expanding dropdowns):
```typescript
function sectionHasActiveLink(pathname: string, section: NavSection): boolean {
  if (section.items) {
    return section.items.some(item => isActive(pathname, item.href))
  }
  if (section.subgroups) {
    return section.subgroups.some(sg =>
      sg.items.some(item => isActive(pathname, item.href))
    )
  }
  return false
}
```

**Component body:**

```typescript
export default function CustomNav() {
  const { config } = useConfig()
  const { user } = useAuth()
  const { navOpen, navRef } = useNav()
  const pathname = usePathname()

  // Dropdown open/close state — keyed by section.key
  const [openSections, setOpenSections] = useState<Record<string, boolean>>({})

  // Initialize dropdown state from URL on mount and pathname changes
  useEffect(() => {
    const newOpen: Record<string, boolean> = {}
    for (const section of [...DROPDOWN_SECTIONS, SYSTEM_SECTION]) {
      newOpen[section.key] = sectionHasActiveLink(pathname, section)
    }
    setOpenSections(newOpen)
  }, [pathname])

  const toggleSection = (key: string) => {
    setOpenSections(prev => ({ ...prev, [key]: !prev[key] }))
  }

  // Shared styles
  const linkBaseClasses = 'flex items-center min-h-[36px] px-4 text-[14px] leading-[1.4] rounded-sm transition-colors duration-150 no-underline'
  const linkInactiveClasses = 'font-normal text-[var(--theme-elevation-600)] hover:text-[var(--theme-elevation-800)] hover:bg-[var(--theme-elevation-50)]'
  const linkActiveClasses = 'font-semibold text-[var(--theme-elevation-900)] bg-[var(--theme-elevation-100)] border-l-[3px] border-l-[var(--theme-success-500)]'

  return (
    <aside className={cn('nav', navOpen && 'nav--nav-open', 'nav--nav-hydrated')}>
      <div className="nav__scroll" ref={navRef}>
        <nav className="nav__wrap flex flex-col py-6">
          {/* Direct Links (NAV-01) */}
          <div className="flex flex-col gap-0.5">
            {DIRECT_LINKS.map(link => {
              const active = isActive(pathname, link.href)
              return (
                <Link
                  key={link.href}
                  href={link.href}
                  className={cn(linkBaseClasses, active ? linkActiveClasses : linkInactiveClasses)}
                  aria-current={active ? 'page' : undefined}
                >
                  {link.label}
                </Link>
              )
            })}
          </div>

          {/* Separator */}
          <div className="my-4 mx-4 border-t border-[var(--theme-elevation-100)]" />

          {/* Dropdown Sections (NAV-02, NAV-03, NAV-04) */}
          {DROPDOWN_SECTIONS.map(section => (
            <Collapsible
              key={section.key}
              open={!!openSections[section.key]}
              onOpenChange={() => toggleSection(section.key)}
            >
              <CollapsibleTrigger className="flex items-center justify-between w-full min-h-[36px] px-4 text-[14px] font-semibold leading-[1.4] text-[var(--theme-elevation-800)] hover:bg-[var(--theme-elevation-50)] transition-colors duration-150 cursor-pointer">
                <span>{section.label}</span>
                <ChevronDown
                  size={16}
                  className={cn(
                    'text-[var(--theme-elevation-400)] transition-transform duration-200',
                    openSections[section.key] && 'rotate-180'
                  )}
                />
              </CollapsibleTrigger>
              <CollapsibleContent>
                {/* Simple items (Bestellungsverwaltung, Website) */}
                {section.items && (
                  <div className="flex flex-col gap-0.5">
                    {section.items.map(item => {
                      const active = isActive(pathname, item.href)
                      return (
                        <Link
                          key={item.href}
                          href={item.href}
                          className={cn(linkBaseClasses, 'pl-8', active ? linkActiveClasses : linkInactiveClasses)}
                          aria-current={active ? 'page' : undefined}
                        >
                          {item.label}
                        </Link>
                      )
                    })}
                  </div>
                )}

                {/* Subgrouped items (Produktverwaltung) — NAV-06 */}
                {section.subgroups && (
                  <div className="flex flex-col">
                    {section.subgroups.map(subgroup => (
                      <div key={subgroup.heading}>
                        <span
                          className="flex items-center min-h-[24px] px-4 pl-8 text-[11px] font-semibold uppercase tracking-[0.05em] leading-[1.2] text-[var(--theme-elevation-450)] mt-3 mb-1 select-none"
                          role="presentation"
                        >
                          {subgroup.heading}
                        </span>
                        <div className="flex flex-col gap-0.5">
                          {subgroup.items.map(item => {
                            const active = isActive(pathname, item.href)
                            return (
                              <Link
                                key={item.href}
                                href={item.href}
                                className={cn(linkBaseClasses, 'pl-12', active ? linkActiveClasses : linkInactiveClasses)}
                                aria-current={active ? 'page' : undefined}
                              >
                                {item.label}
                              </Link>
                            )
                          })}
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </CollapsibleContent>
            </Collapsible>
          ))}

          {/* System Section with WebhookFehlerBadge — NAV-05, INT-01 */}
          <Collapsible
            open={!!openSections[SYSTEM_SECTION.key]}
            onOpenChange={() => toggleSection(SYSTEM_SECTION.key)}
          >
            <CollapsibleTrigger className="flex items-center justify-between w-full min-h-[36px] px-4 text-[14px] font-semibold leading-[1.4] text-[var(--theme-elevation-800)] hover:bg-[var(--theme-elevation-50)] transition-colors duration-150 cursor-pointer">
              <span className="flex items-center gap-2">
                System
                {/* Badge next to System header — visible when dropdown closed (INT-01) */}
                <WebhookFehlerBadge />
              </span>
              <ChevronDown
                size={16}
                className={cn(
                  'text-[var(--theme-elevation-400)] transition-transform duration-200',
                  openSections[SYSTEM_SECTION.key] && 'rotate-180'
                )}
              />
            </CollapsibleTrigger>
            <CollapsibleContent>
              <div className="flex flex-col gap-0.5">
                {SYSTEM_SECTION.items!.map(item => {
                  const active = isActive(pathname, item.href)
                  const isWebhookLink = item.href === '/admin/globals/webhook_errors'
                  return (
                    <Link
                      key={item.href}
                      href={item.href}
                      className={cn(linkBaseClasses, 'pl-8', active ? linkActiveClasses : linkInactiveClasses)}
                      aria-current={active ? 'page' : undefined}
                    >
                      <span className="flex items-center gap-2">
                        {item.label}
                        {/* Badge next to Webhook Fehler link — visible inside dropdown (INT-01) */}
                        {isWebhookLink && <WebhookFehlerBadge />}
                      </span>
                    </Link>
                  )
                })}
              </div>
            </CollapsibleContent>
          </Collapsible>

          {/* Spacer to push logout to bottom */}
          <div className="flex-1" />

          {/* Logout */}
          <div className="px-4 py-4">
            <Logout />
          </div>
        </nav>
      </div>
    </aside>
  )
}
```
